

Test Complete API

# 1. Register a new user
curl -X POST http://localhost:8081/auth/register \
  -H "Content-Type: application/json" \
  -d '{"username":"interview", "email":"interview@test.com", "password":"secure123"}'

# Save the token from response
TOKEN=""  # Copy from above response

# 2. Login with the user
curl -X POST http://localhost:8081/auth/login \
  -H "Content-Type: application/json" \
  -d '{"username":"interview", "password":"secure123"}'

# 3. Create a task (with authentication)
curl -X POST http://localhost:8081/tasks \
  -H "Content-Type: application/json" \
  -H "Authorization: Bearer $TOKEN" \
  -d '{"title":"Prepare for interview", "description":"Review Quarkus and Docker", "status": "PENDING"}'

# 4. Get all tasks
curl -H "Authorization: Bearer $TOKEN" http://localhost:8081/tasks
